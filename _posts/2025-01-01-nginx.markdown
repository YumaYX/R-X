```sh
dnf -y install nginx
systemctl enable --now nginx

dnf -y install firewalld
systemctl enable --now firewalld
firewall-cmd --permanent --add-service=http
firewall-cmd --reload

cp -pv /etc/nginx/nginx.conf /etc/nginx/nginx.conf.bak
```
